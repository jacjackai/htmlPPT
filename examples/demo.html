🚀 快速开始

// 创建应用实例
const app = new HTMLPPT({
    theme: 'light',
    autoSave: true
});

// 添加幻灯片
app.addSlide({
    title: '欢迎使用 HTML PPT',
    content: '这是一个零依赖的演示文稿制作工具'
});

// 进入演示模式
app.present();
