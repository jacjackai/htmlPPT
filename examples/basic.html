🎨 HTML PPT 使用示例

零依赖、纯前端、开箱即用的演示文稿制作工具

📦 基础使用

提示： HTML PPT 可以通过 CDN 或 npm 安装使用

CDN 引入

NPM 安装

npm install htmlppt

import HTMLPPT from 'htmlppt';

🚀 快速开始

// 创建应用实例
const app = new HTMLPPT({
    theme: 'light',
    autoSave: true,
    autoSaveInterval: 30000
});

// 添加幻灯片
app.addSlide({
    title: '欢迎使用 HTML PPT',
    content: '这是一个零依赖的演示文稿制作工具'
});

app.addSlide({
    title: '功能特性',
    content: '• 零依赖\n• 纯前端\n• 开箱即用\n• 自动保存'
});

// 进入演示模式
app.present();

🎯 幻灯片管理

// 添加幻灯片
const slide = app.addSlide({
    title: '新幻灯片',
    content: '幻灯片内容'
});

// 更新幻灯片
app.updateSlide(0, {
    title: '更新后的标题',
    content: '更新后的内容'
});

// 删除幻灯片
app.deleteSlide(0);

// 移动幻灯片
app.moveSlide(0, 2);

// 切换幻灯片
app.switchToSlide(1);

// 获取当前幻灯片
const currentSlide = app.getCurrentSlide();
console.log(currentSlide.title);

🎨 主题管理

// 设置主题
app.setTheme('dark');

// 获取当前主题
const theme = app.getTheme();
console.log(theme); // 'dark'

// 切换主题
app.toggleTheme(); // 切换到 light

// 监听主题变化
app.on('theme:change', (newTheme) => {
    console.log('主题已切换到:', newTheme);
});

🎬 演示模式

// 进入演示模式
app.present(0); // 从第一张幻灯片开始

// 退出演示模式
app.exit();

// 监听演示事件
app.on('present:start', (startIndex) => {
    console.log('演示开始，从第', startIndex, '张幻灯片');
});

app.on('present:end', () => {
    console.log('演示结束');
});

💾 数据持久化

// 手动保存
await app.save();

// 加载保存的数据
await app.load();

// 导出数据
const jsonData = await app.export('json');
const base64Data = await app.export('base64');

// 导入数据
await app.import(jsonData, 'json');
await app.import(base64Data, 'base64');

// 获取分享链接
const shareUrl = app.getShareUrl();
console.log('分享链接:', shareUrl);

↩️ 历史记录

// 撤销操作
app.undo();

// 重做操作
app.redo();

// 检查是否可以撤销/重做
if (app.canUndo()) {
    app.undo();
}

if (app.canRedo()) {
    app.redo();
}

// 监听历史事件
app.on('undo', (state) => {
    console.log('已撤销', state);
});

app.on('redo', (state) => {
    console.log('已重做', state);
});

📡 事件系统

// 监听幻灯片变化
app.on('slide:change', (index, slide) => {
    console.log('切换到幻灯片', index, ':', slide.title);
});

// 监听幻灯片添加
app.on('slide:add', (slide, index) => {
    console.log('添加幻灯片', index, ':', slide.title);
});

// 监听幻灯片删除
app.on('slide:delete', (index) => {
    console.log('删除幻灯片', index);
});

// 监听保存事件
app.on('save', (data) => {
    console.log('数据已保存', data);
});

// 监听加载事件
app.on('load', (data) => {
    console.log('数据已加载', data);
});

// 监听错误事件
app.on('error', (error) => {
    console.error('发生错误:', error);
});

🎯 完整示例

// 创建应用
const app = new HTMLPPT({
    theme: 'light',
    autoSave: true
});

// 等待初始化完成
app.on('ready', () => {
    console.log('应用已就绪');
    
    // 添加幻灯片
    app.addSlide({
        title: '欢迎使用 HTML PPT',
        content: '这是一个零依赖的演示文稿制作工具'
    });
    
    app.addSlide({
        title: '功能特性',
        content: '• 零依赖\n• 纯前端\n• 开箱即用'
    });
    
    // 监听事件
    app.on('slide:change', (index, slide) => {
        console.log('当前幻灯片:', slide.title);
    });
    
    // 开始演示
    setTimeout(() => {
        app.present();
    }, 1000);
});
